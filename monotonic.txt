class Solution {
public:
    bool isMonotonic(vector<int>& nums) {
        int inc = 0;
        int dec = 0;
        int n = nums.size();
        for(int i = 0;i<nums.size()-1;i++){
        if(nums[i]<=nums[i+1]){
            inc++;
        }
        if(nums[i]>=nums[i+1]){
            dec++;
        }
        }
        return inc == n-1|| dec == n-1;
       
    }
};
